  <script>
  (function () {
    const homePage = document.getElementById('home-page');
    const jkPage = document.getElementById('jk-page');

    function showView(view, push = true) {
      document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
      if (view === 'jk') {
        jkPage.classList.add('active');
        document.title = 'Zenith Travel | Jammu & Kashmir';
        if (push) history.pushState({page:'jk'}, 'Jammu & Kashmir', '#jk');
        const v = document.getElementById('jk-video-bg'); if (v && v.paused) v.play().catch(()=>{});
      } else {
        homePage.classList.add('active');
        document.title = 'Zenith Travel Agency | Reach Your Peak Adventure';
        if (push) history.pushState({page:'home'}, 'Home', '#home');
        const v = document.getElementById('jk-video-bg'); if (v && !v.paused) { v.pause(); v.currentTime = 0; }
      }
      window.scrollTo({top:0,behavior:'instant'});
    }

    // Click handling: target switching by data-target OR href fragment
    document.addEventListener('click', function (e) {
      const a = e.target.closest('a');
      if (!a) return;

      // Back link
      if (a.hasAttribute('data-back')) {
        e.preventDefault();
        showView('home');
        return;
      }

      // nav links that scroll to home sections
      const navName = a.getAttribute('data-nav');
      if (navName) {
        e.preventDefault();
        showView('home');
        setTimeout(() => {
          const sec = document.getElementById(navName);
          if (sec) sec.scrollIntoView({behavior:'smooth'});
        }, 80);
        return;
      }

      // contact/cta scroll
      if (a.getAttribute('data-scroll') === 'contact') {
        e.preventDefault();
        showView('home');
        setTimeout(()=> {
          const sec = document.getElementById('contact');
          if (sec) sec.scrollIntoView({behavior:'smooth'});
        }, 80);
        return;
      }

      // Location cards: check data-target or href fragment
      const target = a.getAttribute('data-target') || a.getAttribute('href') || '';
      if (a.dataset.target === 'jk' || target.includes('#jk') || target.includes('jammu-kashmir')) {
        e.preventDefault();
        showView('jk');
        return;
      }
      if (a.dataset.target === 'kerala' || target.includes('#kerala')) {
        e.preventDefault();
        // (No separate view for Kerala yet) - scroll to home and highlight if desired
        showView('home');
        setTimeout(()=> {
          const sec = document.getElementById('locations');
          if (sec) sec.scrollIntoView({behavior:'smooth'});
        }, 80);
        return;
      }
      if (a.dataset.target === 'rajasthan' || target.includes('#rajasthan')) {
        e.preventDefault();
        showView('home');
        setTimeout(()=> {
          const sec = document.getElementById('locations');
          if (sec) sec.scrollIntoView({behavior:'smooth'});
        }, 80);
        return;
      }
    });

    // Handle browser back/forward
    window.addEventListener('popstate', function () {
      const hash = location.hash || '';
      if (hash === '#jk') {
        showView('jk', false);
      } else {
        showView('home', false);
      }
    });

    // Initial load: respect hash if present
    if (location.hash === '#jk') {
      showView('jk', false);
    } else {
      showView('home', false);
    }
  })();
  </script>
